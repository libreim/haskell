\begin{frame}[fragile]{Quicksort}
  Implementación del algoritmo Quicksort
  \begin{lstlisting}[language=haskell]
qsort []     = []
qsort (x:xs) = qsort [y | y<-xs, y<=x]
            ++ [x]
            ++ qsort [y | y<-xs, y>x]
  \end{lstlisting}
\end{frame}
